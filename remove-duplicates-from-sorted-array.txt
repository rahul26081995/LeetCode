https://leetcode.com/problems/remove-duplicates-from-sorted-array/

class Solution {
public:
    int removeDuplicates(vector<int>& nums) {
        int n = nums.size();
        int j=0;
        for(int i=0;i<n;i++){
            nums[j]=nums[i];
            while(i<n-1 && nums[i]==nums[i+1]){
                i++;
            }
            j++;
        }
          return j;
    }
  
};
